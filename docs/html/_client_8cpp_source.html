370  QNetworkConfigurationManager manager;

372 
373  // Get saved network configuration
374  QSettings settings(QSettings::UserScope, QLatin1String("QtProject"));
375  settings.beginGroup(QLatin1String("QtNetwork"));

377  settings.endGroup();
378
